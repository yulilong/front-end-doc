# Mapе’ҢWeakMap

# 1. Mapеҹәжң¬з”Ёжі•

JSзҡ„еҜ№иұЎпјҢжң¬иҙЁдёҠеҖјй”®еҖјеҜ№зҡ„йӣҶеҗҲ(Hashз»“жһ„)пјҢдҪҶжҳҜй”®еҸӘиғҪжҳҜеӯ—з¬ҰдёІпјҢиҝҷз»ҷе®ғзҡ„дҪҝз”ЁеёҰжқҘдәҶеҫҲеӨ§йҷҗеҲ¶гҖӮ

дёәдәҶи§ЈеҶіиҝҷдёӘй—®йўҳпјҢES6 жҸҗдҫӣдәҶ Map ж•°жҚ®з»“жһ„гҖӮе®ғзұ»дјјдәҺеҜ№иұЎпјҢд№ҹжҳҜй”®еҖјеҜ№зҡ„йӣҶеҗҲпјҢдҪҶжҳҜвҖңй”®вҖқзҡ„иҢғеӣҙдёҚйҷҗдәҺеӯ—з¬ҰдёІпјҢеҗ„з§Қзұ»еһӢзҡ„еҖјпјҲеҢ…жӢ¬еҜ№иұЎпјүйғҪеҸҜд»ҘеҪ“дҪңй”®гҖӮд№ҹе°ұжҳҜиҜҙпјҢObject з»“жһ„жҸҗдҫӣдәҶвҖңеӯ—з¬ҰдёІвҖ”еҖјвҖқзҡ„еҜ№еә”пјҢMap з»“жһ„жҸҗдҫӣдәҶвҖңеҖјвҖ”еҖјвҖқзҡ„еҜ№еә”пјҢжҳҜдёҖз§Қжӣҙе®Ңе–„зҡ„ Hash з»“жһ„е®һзҺ°гҖӮеҰӮжһңдҪ йңҖиҰҒвҖңй”®еҖјеҜ№вҖқзҡ„ж•°жҚ®з»“жһ„пјҢMap жҜ” Object жӣҙеҗҲйҖӮгҖӮ

var e = {a: 10}
var data = {}
// з”ұдәҺеҜ№иұЎеҸӘжҺҘеҸ—еӯ—з¬ҰдёІдҪңдёәй”®еҗҚпјҢжүҖд»Ҙeиў«иҮӘеҠЁиҪ¬дёәеӯ—з¬ҰдёІ[object Object]
data[e] = 300
data // {[object Object]: 200}

const m = new Map();
const o = {p: 'Hello World'};
// еҗ‘ Map ж·»еҠ жҲҗе‘ҳ
m.set(o, 'content')
m.get(o) // "content"
m.has(o) // true
m.delete(o) // true
m.has(o) // false

дҪңдёәжһ„йҖ еҮҪж•°пјҢMap д№ҹеҸҜд»ҘжҺҘеҸ—дёҖдёӘж•°з»„дҪңдёәеҸӮж•°гҖӮиҜҘж•°з»„зҡ„жҲҗе‘ҳжҳҜдёҖдёӘдёӘиЎЁзӨәй”®еҖјеҜ№зҡ„ж•°з»„гҖӮдәӢе®һдёҠпјҢдёҚд»…д»…жҳҜж•°з»„пјҢд»»дҪ•е…·жңү Iterator жҺҘеҸЈгҖҒдё”жҜҸдёӘжҲҗе‘ҳйғҪжҳҜдёҖдёӘеҸҢе…ғзҙ зҡ„ж•°з»„зҡ„ж•°жҚ®з»“жһ„пјҲиҜҰи§ҒгҖҠIteratorгҖӢдёҖз« пјүйғҪеҸҜд»ҘеҪ“дҪңMapжһ„йҖ еҮҪж•°зҡ„еҸӮж•°гҖӮиҝҷе°ұжҳҜиҜҙпјҢSetе’ҢMapйғҪеҸҜд»Ҙз”ЁжқҘз”ҹжҲҗж–°зҡ„ MapгҖӮ

const map = new Map([ ['name', 'еј дёү'], ['title', 'Author'] ]);
// зӣёеҪ“дәҺ
// items = [ ['name', 'еј дёү'], ['title', 'Author'] ];
// const map = new Map();
// items.forEach( ([key, value]) => map.set(key, value) );
map.size // 2
map.has('name') // true
map.get('name') // "еј дёү"

const set = new Set([ ['foo', 1], ['bar', 2] ]);
const m1 = new Map(set);
m1.get('foo') // 1
const m2 = new Map([['baz', 3]]);
const m3 = new Map(m2);
m3.get('baz') // 3

еҰӮжһңеҜ№еҗҢдёҖдёӘй”®еӨҡж¬ЎиөӢеҖјпјҢеҗҺйқўзҡ„еҖје°ҶиҰҶзӣ–еүҚйқўзҡ„еҖјгҖӮеҰӮжһңиҜ»еҸ–дёҖдёӘжңӘзҹҘзҡ„й”®пјҢеҲҷиҝ”еӣһundefinedгҖӮ

var map = new Map();
map.set(1, 'aaa').set(1, 'bbb');
map.get(1) // "bbb"
map.get('asfddfsasadf') // undefined

жіЁж„ҸпјҢеҸӘжңүеҜ№еҗҢдёҖдёӘеҜ№иұЎзҡ„еј•з”ЁпјҢMap з»“жһ„жүҚе°Ҷе…¶и§ҶдёәеҗҢдёҖдёӘй”®гҖӮиҝҷдёҖзӮ№иҰҒйқһеёёе°ҸеҝғгҖӮеҗҢзҗҶпјҢеҗҢж ·зҡ„еҖјзҡ„дёӨдёӘе®һдҫӢпјҢеңЁ Map з»“жһ„дёӯиў«и§ҶдёәдёӨдёӘй”®гҖӮ

з”ұдёҠеҸҜзҹҘпјҢMap зҡ„й”®е®һйҷ…дёҠжҳҜи·ҹеҶ…еӯҳең°еқҖз»‘е®ҡзҡ„пјҢеҸӘиҰҒеҶ…еӯҳең°еқҖдёҚдёҖж ·пјҢе°ұи§ҶдёәдёӨдёӘй”®гҖӮиҝҷе°ұи§ЈеҶідәҶеҗҢеҗҚеұһжҖ§зў°ж’һпјҲclashпјүзҡ„й—®йўҳпјҢжҲ‘д»¬жү©еұ•еҲ«дәәзҡ„еә“зҡ„ж—¶еҖҷпјҢеҰӮжһңдҪҝз”ЁеҜ№иұЎдҪңдёәй”®еҗҚпјҢе°ұдёҚз”ЁжӢ…еҝғиҮӘе·ұзҡ„еұһжҖ§дёҺеҺҹдҪңиҖ…зҡ„еұһжҖ§еҗҢеҗҚгҖӮ

еҰӮжһң Map зҡ„й”®жҳҜдёҖдёӘз®ҖеҚ•зұ»еһӢзҡ„еҖјпјҲж•°еӯ—гҖҒеӯ—з¬ҰдёІгҖҒеёғе°”еҖјпјүпјҢеҲҷеҸӘиҰҒдёӨдёӘеҖјдёҘж јзӣёзӯүпјҢMap е°Ҷе…¶и§ҶдёәдёҖдёӘй”®пјҢжҜ”еҰӮ0е’Ң-0е°ұжҳҜдёҖдёӘй”®пјҢеёғе°”еҖјtrueе’Ңеӯ—з¬ҰдёІtrueеҲҷжҳҜдёӨдёӘдёҚеҗҢзҡ„й”®гҖӮеҸҰеӨ–пјҢundefinedе’Ңnullд№ҹжҳҜдёӨдёӘдёҚеҗҢзҡ„й”®гҖӮиҷҪз„¶NaNдёҚдёҘж јзӣёзӯүдәҺиҮӘиә«пјҢдҪҶ Map е°Ҷе…¶и§ҶдёәеҗҢдёҖдёӘй”®гҖӮ

// еҗҢж ·зҡ„еҖјзҡ„дёӨдёӘе®һдҫӢпјҢеңЁ Map з»“жһ„дёӯиў«и§ҶдёәдёӨдёӘй”®
var map = new Map(); var k1 = ['a']; var k2 = ['a'];
map .set(k1, 111) .set(k2, 222);
map.get(k1) // 111
map.get(k2) // 222

map.set(-0, 123);
map.get(+0) // 123

map.set(true, 1);
map.set('true', 2);
map.get(true) // 1

map.set(undefined, 3);
map.set(null, 4);
map.get(undefined) // 3

map.set(NaN, 123);
map.get(NaN) // 123

# 2. Mapе®һдҫӢзҡ„еұһжҖ§е’Ңж“ҚдҪңж–№жі•

# 2.1 sizeеұһжҖ§

sizeеұһжҖ§иҝ”еӣһ Map з»“жһ„зҡ„жҲҗе‘ҳжҖ»ж•°гҖӮ

var map = new Map();
map.set('foo', true);
map.set('bar', false);
map.size // 2

# 2.2 Map.prototype.set(key, value)

setж–№жі•и®ҫзҪ®й”®еҗҚkeyеҜ№еә”зҡ„й”®еҖјдёәvalueпјҢз„¶еҗҺиҝ”еӣһж•ҙдёӘ Map з»“жһ„гҖӮеҰӮжһңkeyе·Із»ҸжңүеҖјпјҢеҲҷй”®еҖјдјҡиў«жӣҙж–°пјҢеҗҰеҲҷе°ұж–°з”ҹжҲҗиҜҘй”®гҖӮsetж–№жі•иҝ”еӣһзҡ„жҳҜеҪ“еүҚзҡ„MapеҜ№иұЎпјҢеӣ жӯӨеҸҜд»ҘйҮҮз”Ёй“ҫејҸеҶҷжі•гҖӮ

const m = new Map();

m.set('edition', 6)        // й”®жҳҜеӯ—з¬ҰдёІ
m.set(262, 'standard')     // й”®жҳҜж•°еҖј
m.set(undefined, 'nah')    // й”®жҳҜ undefined
let map = new Map()
  .set(1, 'a')
  .set(2, 'b')
  .set(3, 'c');

# 2.3 Map.prototype.get(key)

getж–№жі•иҜ»еҸ–keyеҜ№еә”зҡ„й”®еҖјпјҢеҰӮжһңжүҫдёҚеҲ°keyпјҢиҝ”еӣһundefinedгҖӮ

var m = new Map();
var hello = function() {console.log('hello');};
m.set(hello, 'Hello ES6!') // й”®жҳҜеҮҪж•°
m.get(hello)  // Hello ES6!

# 2.4 Map.prototype.has(key)

hasж–№жі•иҝ”еӣһдёҖдёӘеёғе°”еҖјпјҢиЎЁзӨәжҹҗдёӘй”®жҳҜеҗҰеңЁеҪ“еүҚ Map еҜ№иұЎд№ӢдёӯгҖӮ

var m = new Map();
m.set('edition', 6);
m.set(262, 'standard');
m.set(undefined, 'nah');

m.has('edition')     // true
m.has('years')       // false
m.has(262)           // true
m.has(undefined)     // true

# 2.5 Map.prototype.delete(key)

deleteж–№жі•еҲ йҷӨжҹҗдёӘй”®пјҢиҝ”еӣһtrueгҖӮеҰӮжһңеҲ йҷӨеӨұиҙҘпјҢиҝ”еӣһfalseгҖӮ

const m = new Map();
m.set(undefined, 'nah');
m.has(undefined)     // true

m.delete(undefined)
m.has(undefined)       // false

# 2.6 Map.prototype.clear()

clearж–№жі•жё…йҷӨжүҖжңүжҲҗе‘ҳпјҢжІЎжңүиҝ”еӣһеҖјгҖӮ

let map = new Map();
map.set('foo', true);
map.set('bar', false);

map.size // 2
map.clear()
map.size // 0

# 3. MapйҒҚеҺҶж–№жі•

Map з»“жһ„еҺҹз”ҹжҸҗдҫӣдёүдёӘйҒҚеҺҶеҷЁз”ҹжҲҗеҮҪж•°е’ҢдёҖдёӘйҒҚеҺҶж–№жі•гҖӮ

  • Map.prototype.keys()пјҡиҝ”еӣһй”®еҗҚзҡ„йҒҚеҺҶеҷЁгҖӮ
  • Map.prototype.values()пјҡиҝ”еӣһй”®еҖјзҡ„йҒҚеҺҶеҷЁгҖӮ
  • Map.prototype.entries()пјҡиҝ”еӣһжүҖжңүжҲҗе‘ҳзҡ„йҒҚеҺҶеҷЁгҖӮ
  • Map.prototype.forEach()пјҡйҒҚеҺҶ Map зҡ„жүҖжңүжҲҗе‘ҳгҖӮ

йңҖиҰҒзү№еҲ«жіЁж„Ҹзҡ„жҳҜпјҢMap зҡ„йҒҚеҺҶйЎәеәҸе°ұжҳҜжҸ’е…ҘйЎәеәҸгҖӮMap з»“жһ„зҡ„й»ҳи®ӨйҒҚеҺҶеҷЁжҺҘеҸЈпјҲSymbol.iteratorеұһжҖ§пјүпјҢе°ұжҳҜentriesж–№жі•гҖӮ

var map = new Map([ ['F', 'no'], ['T',  'yes'], ]);

for (let key of map.keys()) { console.log(key); }
// "F" "T"
for (let value of map.values()) { console.log(value); }
// "no" "yes"
for (let [key, value] of map.entries()) { console.log(key, value); }
// "F" "no"  "T" "yes"
// зӯүеҗҢдәҺдҪҝз”Ёmap.entries()
for (let [key, value] of map) { console.log(key, value); }
// "F" "no" "T" "yes"

map[Symbol.iterator] === map.entries
// true


// з»“еҗҲж•°з»„зҡ„mapж–№жі•гҖҒfilterж–№жі•пјҢеҸҜд»Ҙе®һзҺ° Map зҡ„йҒҚеҺҶе’ҢиҝҮж»ӨпјҲMap жң¬иә«жІЎжңүmapе’Ңfilterж–№жі•пјүгҖӮ
var map0 = new Map() .set(1, 'a') .set(2, 'b') .set(3, 'c');
var map1 = new Map([...map0].filter(([k, v]) => k < 3) );
// дә§з”ҹ Map з»“жһ„ {1 => 'a', 2 => 'b'}
var map2 = new Map( [...map0].map(([k, v]) => [k * 2, '_' + v]) );
// дә§з”ҹ Map з»“жһ„ {2 => '_a', 4 => '_b', 6 => '_c'}

жӯӨеӨ–пјҢMap иҝҳжңүдёҖдёӘforEachж–№жі•пјҢдёҺж•°з»„зҡ„forEachж–№жі•зұ»дјјпјҢд№ҹеҸҜд»Ҙе®һзҺ°йҒҚеҺҶгҖӮforEachж–№жі•иҝҳеҸҜд»ҘжҺҘеҸ—з¬¬дәҢдёӘеҸӮж•°пјҢз”ЁжқҘз»‘е®ҡthisгҖӮ

var map = new Map() .set(1, 'a') .set(2, 'b') .set(3, 'c');
map.forEach(function(value, key, map) {
  console.log("Key: %s, Value: %s", key, value);
});
const reporter = {
  report: function(key, value) { console.log("Key: %s, Value: %s", key, value); }
};
map.forEach(function(value, key, map) { this.report(key, value); }, reporter);

# 4. MapдёҺе…¶д»–ж•°жҚ®з»“жһ„зҡ„дә’зӣёиҪ¬жҚў

# 4.1 MapиҪ¬дёәж•°з»„

Map иҪ¬дёәж•°з»„жңҖж–№дҫҝзҡ„ж–№жі•пјҢе°ұжҳҜдҪҝз”Ёжү©еұ•иҝҗз®—з¬ҰпјҲ...пјүгҖӮ

var myMap = new Map().set(true, 7).set({foo: 3}, ['abc']);
[...myMap]
// [ [ true, 7 ], [ { foo: 3 }, [ 'abc' ] ] ]

# 4.2 ж•°з»„иҪ¬дёәMap

е°Ҷж•°з»„дј е…Ҙ Map жһ„йҖ еҮҪж•°пјҢе°ұеҸҜд»ҘиҪ¬дёә MapгҖӮ

new Map([ [true, 7], [{foo: 3}, ['abc']] ])
// Map(2) {true => 7, {вҖҰ} => Array(1)}

# 4.3 MapиҪ¬дёәеҜ№иұЎ

еҰӮжһңжүҖжңү Map зҡ„й”®йғҪжҳҜеӯ—з¬ҰдёІпјҢе®ғеҸҜд»Ҙж— жҚҹең°иҪ¬дёәеҜ№иұЎгҖӮMapеҰӮжһңжңүйқһеӯ—з¬ҰдёІзҡ„й”®еҗҚпјҢйӮЈд№ҲиҝҷдёӘй”®еҗҚдјҡиў«иҪ¬жҲҗеӯ—з¬ҰдёІпјҢеҶҚдҪңдёәеҜ№иұЎзҡ„й”®еҗҚгҖӮ

function strMapToObj(strMap) {
  let obj = Object.create(null);
  for (let [k,v] of strMap) { obj[k] = v; }
  return obj;
}
var myMap = new Map().set('yes', true).set('no', false);
strMapToObj(myMap)
// { yes: true, no: false }

# 4.4 еҜ№иұЎиҪ¬дёәMap

еҜ№иұЎиҪ¬дёә Map еҸҜд»ҘйҖҡиҝҮObject.entries()гҖӮ

let obj = {"a":1, "b":2};
let map = new Map(Object.entries(obj));

жӯӨеӨ–пјҢд№ҹеҸҜд»ҘиҮӘе·ұе®һзҺ°дёҖдёӘиҪ¬жҚўеҮҪж•°гҖӮ

function objToStrMap(obj) {
  let strMap = new Map();
  for (let k of Object.keys(obj)) { strMap.set(k, obj[k]); }
  return strMap;
}
objToStrMap({yes: true, no: false})
// Map {"yes" => true, "no" => false}

# 4.5 Map иҪ¬дёә JSON

Map иҪ¬дёә JSON иҰҒеҢәеҲҶдёӨз§Қжғ…еҶөгҖӮдёҖз§Қжғ…еҶөжҳҜпјҢMap зҡ„й”®еҗҚйғҪжҳҜеӯ—з¬ҰдёІпјҢиҝҷж—¶еҸҜд»ҘйҖүжӢ©иҪ¬дёәеҜ№иұЎ JSONгҖӮ

function strMapToJson(strMap) { return JSON.stringify(strMapToObj(strMap)); }

let myMap = new Map().set('yes', true).set('no', false);
strMapToJson(myMap)
// '{"yes":true,"no":false}'

еҸҰдёҖз§Қжғ…еҶөжҳҜпјҢMap зҡ„й”®еҗҚжңүйқһеӯ—з¬ҰдёІпјҢиҝҷж—¶еҸҜд»ҘйҖүжӢ©иҪ¬дёәж•°з»„ JSONгҖӮ

function mapToArrayJson(map) { return JSON.stringify([...map]); }

let myMap = new Map().set(true, 7).set({foo: 3}, ['abc']);
mapToArrayJson(myMap)
// '[[true,7],[{"foo":3},["abc"]]]'

# 4.6 JSON иҪ¬дёә Map

JSON иҪ¬дёә MapпјҢжӯЈеёёжғ…еҶөдёӢпјҢжүҖжңүй”®еҗҚйғҪжҳҜеӯ—з¬ҰдёІгҖӮ

function jsonToStrMap(jsonStr) { return objToStrMap(JSON.parse(jsonStr)); }

jsonToStrMap('{"yes": true, "no": false}')
// Map {'yes' => true, 'no' => false}

дҪҶжҳҜпјҢжңүдёҖз§Қзү№ж®Ҡжғ…еҶөпјҢж•ҙдёӘ JSON е°ұжҳҜдёҖдёӘж•°з»„пјҢдё”жҜҸдёӘж•°з»„жҲҗе‘ҳжң¬иә«пјҢеҸҲжҳҜдёҖдёӘжңүдёӨдёӘжҲҗе‘ҳзҡ„ж•°з»„гҖӮиҝҷж—¶пјҢе®ғеҸҜд»ҘдёҖдёҖеҜ№еә”ең°иҪ¬дёә MapгҖӮиҝҷеҫҖеҫҖжҳҜ Map иҪ¬дёәж•°з»„ JSON зҡ„йҖҶж“ҚдҪңгҖӮ

function jsonToMap(jsonStr) { return new Map(JSON.parse(jsonStr)); }

jsonToMap('[[true,7],[{"foo":3},["abc"]]]')
// Map {true => 7, Object {foo: 3} => ['abc']}

# 5. WeakMap

WeakMapз»“жһ„дёҺMapз»“жһ„зұ»дјјпјҢд№ҹжҳҜз”ЁдәҺз”ҹжҲҗй”®еҖјеҜ№зҡ„йӣҶеҗҲгҖӮWeakMapдёҺMapзҡ„еҢәеҲ«жңүдёӨзӮ№гҖӮ

йҰ–е…ҲпјҢWeakMapеҸӘжҺҘеҸ—еҜ№иұЎдҪңдёәй”®еҗҚпјҲnullйҷӨеӨ–пјүпјҢдёҚжҺҘеҸ—е…¶д»–зұ»еһӢзҡ„еҖјдҪңдёәй”®еҗҚгҖӮ

е…¶ж¬ЎпјҢWeakMapзҡ„й”®еҗҚжүҖжҢҮеҗ‘зҡ„еҜ№иұЎпјҢдёҚи®Ўе…Ҙеһғеңҫеӣһж”¶жңәеҲ¶гҖӮ

WeakMapзҡ„и®ҫи®Ўзӣ®зҡ„еңЁдәҺпјҢжңүж—¶жҲ‘д»¬жғіеңЁжҹҗдёӘеҜ№иұЎдёҠйқўеӯҳж”ҫдёҖдәӣж•°жҚ®пјҢдҪҶжҳҜиҝҷдјҡеҪўжҲҗеҜ№дәҺиҝҷдёӘеҜ№иұЎзҡ„еј•з”ЁгҖӮдёҖж—ҰдёҚеҶҚйңҖиҰҒиҝҷдёӘеҜ№иұЎпјҢжҲ‘д»¬е°ұеҝ…йЎ»жүӢеҠЁеҲ йҷӨиҝҷдёӘеј•з”ЁпјҢеҗҰеҲҷеһғеңҫеӣһж”¶жңәеҲ¶е°ұдёҚдјҡйҮҠж”ҫеҜ№иұЎеҚ з”Ёзҡ„еҶ…еӯҳгҖӮ

WeakMap е°ұжҳҜдёәдәҶи§ЈеҶіиҝҷдёӘй—®йўҳиҖҢиҜһз”ҹзҡ„пјҢе®ғзҡ„й”®еҗҚжүҖеј•з”Ёзҡ„еҜ№иұЎйғҪжҳҜејұеј•з”ЁпјҢеҚіеһғеңҫеӣһж”¶жңәеҲ¶дёҚе°ҶиҜҘеј•з”ЁиҖғиҷ‘еңЁеҶ…гҖӮеӣ жӯӨпјҢеҸӘиҰҒжүҖеј•з”Ёзҡ„еҜ№иұЎзҡ„е…¶д»–еј•з”ЁйғҪиў«жё…йҷӨпјҢеһғеңҫеӣһж”¶жңәеҲ¶е°ұдјҡйҮҠж”ҫиҜҘеҜ№иұЎжүҖеҚ з”Ёзҡ„еҶ…еӯҳгҖӮд№ҹе°ұжҳҜиҜҙпјҢдёҖж—ҰдёҚеҶҚйңҖиҰҒпјҢWeakMap йҮҢйқўзҡ„й”®еҗҚеҜ№иұЎе’ҢжүҖеҜ№еә”зҡ„й”®еҖјеҜ№дјҡиҮӘеҠЁж¶ҲеӨұпјҢдёҚз”ЁжүӢеҠЁеҲ йҷӨеј•з”ЁгҖӮ

жҖ»д№ӢпјҢWeakMapзҡ„дё“з”ЁеңәеҗҲе°ұжҳҜпјҢе®ғзҡ„й”®жүҖеҜ№еә”зҡ„еҜ№иұЎпјҢеҸҜиғҪдјҡеңЁе°ҶжқҘж¶ҲеӨұгҖӮWeakMapз»“жһ„жңүеҠ©дәҺйҳІжӯўеҶ…еӯҳжі„жјҸгҖӮ

жіЁж„ҸпјҢWeakMap ејұеј•з”Ёзҡ„еҸӘжҳҜй”®еҗҚпјҢиҖҢдёҚжҳҜй”®еҖјгҖӮй”®еҖјдҫқз„¶жҳҜжӯЈеёёеј•з”ЁгҖӮ

const wm = new WeakMap();
let key = {};
let obj = {foo: 1};

wm.set(key, obj);
obj = null;
wm.get(key)
// Object {foo: 1}

дёҠйқўд»Јз ҒдёӯпјҢй”®еҖјobjжҳҜжӯЈеёёеј•з”ЁгҖӮжүҖд»ҘпјҢеҚідҪҝеңЁ WeakMap еӨ–йғЁж¶ҲйҷӨдәҶobjзҡ„еј•з”ЁпјҢWeakMap еҶ…йғЁзҡ„еј•з”Ёдҫқз„¶еӯҳеңЁгҖӮ

# 6. WeakMapиҜӯжі•

WeakMap дёҺ Map еңЁ API дёҠзҡ„еҢәеҲ«дё»иҰҒжҳҜдёӨдёӘпјҢдёҖжҳҜжІЎжңүйҒҚеҺҶж“ҚдҪңпјҲеҚіжІЎжңүkeys()гҖҒvalues()е’Ңentries()ж–№жі•пјүпјҢд№ҹжІЎжңүsizeеұһжҖ§гҖӮеӣ дёәжІЎжңүеҠһжі•еҲ—еҮәжүҖжңүй”®еҗҚпјҢжҹҗдёӘй”®еҗҚжҳҜеҗҰеӯҳеңЁе®Ңе…ЁдёҚеҸҜйў„жөӢпјҢи·ҹеһғеңҫеӣһж”¶жңәеҲ¶жҳҜеҗҰиҝҗиЎҢзӣёе…ігҖӮиҝҷдёҖеҲ»еҸҜд»ҘеҸ–еҲ°й”®еҗҚпјҢдёӢдёҖеҲ»еһғеңҫеӣһж”¶жңәеҲ¶зӘҒз„¶иҝҗиЎҢдәҶпјҢиҝҷдёӘй”®еҗҚе°ұжІЎдәҶпјҢдёәдәҶйҳІжӯўеҮәзҺ°дёҚзЎ®е®ҡжҖ§пјҢе°ұз»ҹдёҖи§„е®ҡдёҚиғҪеҸ–еҲ°й”®еҗҚгҖӮдәҢжҳҜж— жі•жё…з©әпјҢеҚідёҚж”ҜжҢҒclearж–№жі•гҖӮеӣ жӯӨпјҢWeakMapеҸӘжңүеӣӣдёӘж–№жі•еҸҜз”Ёпјҡget()гҖҒset()гҖҒhas()гҖҒdelete()гҖӮ

const wm = new WeakMap();

// sizeгҖҒforEachгҖҒclear ж–№жі•йғҪдёҚеӯҳеңЁ
wm.size // undefined
wm.forEach // undefined
wm.clear // undefined

# 7. WeakMap зҡ„з”ЁйҖ”

еүҚж–ҮиҜҙиҝҮпјҢWeakMap еә”з”Ёзҡ„е…ёеһӢеңәеҗҲе°ұжҳҜ DOM иҠӮзӮ№дҪңдёәй”®еҗҚгҖӮдёӢйқўжҳҜдёҖдёӘдҫӢеӯҗгҖӮ

let myWeakmap = new WeakMap();

myWeakmap.set(
  document.getElementById('logo'),
  {timesClicked: 0})
;

document.getElementById('logo').addEventListener('click', function() {
  let logoData = myWeakmap.get(document.getElementById('logo'));
  logoData.timesClicked++;
}, false);

дёҠйқўд»Јз ҒдёӯпјҢdocument.getElementById('logo')жҳҜдёҖдёӘ DOM иҠӮзӮ№пјҢжҜҸеҪ“еҸ‘з”ҹclickдәӢд»¶пјҢе°ұжӣҙж–°дёҖдёӢзҠ¶жҖҒгҖӮжҲ‘д»¬е°ҶиҝҷдёӘзҠ¶жҖҒдҪңдёәй”®еҖјж”ҫеңЁ WeakMap йҮҢпјҢеҜ№еә”зҡ„й”®еҗҚе°ұжҳҜиҝҷдёӘиҠӮзӮ№еҜ№иұЎгҖӮдёҖж—ҰиҝҷдёӘ DOM иҠӮзӮ№еҲ йҷӨпјҢиҜҘзҠ¶жҖҒе°ұдјҡиҮӘеҠЁж¶ҲеӨұпјҢдёҚеӯҳеңЁеҶ…еӯҳжі„жјҸйЈҺйҷ©гҖӮ

WeakMap зҡ„еҸҰдёҖдёӘз”ЁеӨ„жҳҜйғЁзҪІз§ҒжңүеұһжҖ§гҖӮ

const _counter = new WeakMap();
const _action = new WeakMap();

class Countdown {
  constructor(counter, action) {
    _counter.set(this, counter);
    _action.set(this, action);
  }
  dec() {
    let counter = _counter.get(this);
    if (counter < 1) return;
    counter--;
    _counter.set(this, counter);
    if (counter === 0) {
      _action.get(this)();
    }
  }
}

const c = new Countdown(2, () => console.log('DONE'));

c.dec()
c.dec()
// DONE

дёҠйқўд»Јз ҒдёӯпјҢCountdownзұ»зҡ„дёӨдёӘеҶ…йғЁеұһжҖ§_counterе’Ң_actionпјҢжҳҜе®һдҫӢзҡ„ејұеј•з”ЁпјҢжүҖд»ҘеҰӮжһңеҲ йҷӨе®һдҫӢпјҢе®ғд»¬д№ҹе°ұйҡҸд№Ӣж¶ҲеӨұпјҢдёҚдјҡйҖ жҲҗеҶ…еӯҳжі„жјҸгҖӮ

# Object еҜ№иұЎ - ж–°еўһж–№жі•

иҪ¬иҪҪж•ҙзҗҶиҮӘпјҡhttps://es6.ruanyifeng.com/#docs/object-methods

# 1. Object.is()

Object.isз”ЁжқҘжҜ”иҫғдёӨдёӘеҖјжҳҜеҗҰдёҘж јзӣёзӯүпјҢдёҺдёҘж јжҜ”иҫғиҝҗз®—з¬ҰпјҲ===пјүзҡ„иЎҢдёәеҹәжң¬дёҖиҮҙгҖӮдёҚеҗҢд№ӢеӨ„еҸӘжңүдёӨдёӘпјҡдёҖжҳҜ+0дёҚзӯүдәҺ-0пјҢдәҢжҳҜNaNзӯүдәҺиҮӘиә«гҖӮ

ES5 жҜ”иҫғдёӨдёӘеҖјжҳҜеҗҰзӣёзӯүпјҢеҸӘжңүдёӨдёӘиҝҗз®—з¬Ұпјҡ

  • ==пјҡзӣёзӯүиҝҗз®—з¬ҰпјҢзјәзӮ№пјҡдјҡиҮӘеҠЁиҪ¬жҚўж•°жҚ®зұ»еһӢ
  • ===пјҡдёҘж јзӣёзӯүиҝҗз®—з¬ҰпјҢзјәзӮ№пјҡNaNдёҚзӯүдәҺиҮӘиә«пјҢд»ҘеҸҠ+0зӯүдәҺ-0

JavaScript зјәд№ҸдёҖз§Қиҝҗз®—пјҢеңЁжүҖжңүзҺҜеўғдёӯпјҢеҸӘиҰҒдёӨдёӘеҖјжҳҜдёҖж ·зҡ„пјҢе®ғд»¬е°ұеә”иҜҘзӣёзӯүгҖӮES6 жҸҗеҮәвҖңSame-value equalityвҖқпјҲеҗҢеҖјзӣёзӯүпјүз®—жі•пјҢз”ЁжқҘи§ЈеҶіиҝҷдёӘй—®йўҳгҖӮObject.isе°ұжҳҜйғЁзҪІиҝҷдёӘз®—жі•зҡ„ж–°ж–№жі•гҖӮ

Object.is('foo', 'foo') // true
Object.is({}, {}) // false

+0 === -0 //true
NaN === NaN // false

Object.is(+0, -0) // false
Object.is(NaN, NaN) // true

ES5 еҸҜд»ҘйҖҡиҝҮдёӢйқўзҡ„д»Јз ҒпјҢйғЁзҪІObject.isгҖӮ

Object.defineProperty(Object, 'is', {
  value: function(x, y) {
    if (x === y) {
      // й’ҲеҜ№+0 дёҚзӯүдәҺ -0зҡ„жғ…еҶө
      return x !== 0 || 1 / x === 1 / y;
    }
    // й’ҲеҜ№NaNзҡ„жғ…еҶө
    return x !== x && y !== y;
  },
  configurable: true,
  enumerable: false,
  writable: true
});

# 2. Object.assign()

Object.assignж–№жі•з”ЁдәҺеҜ№иұЎзҡ„еҗҲе№¶пјҢе°ҶжәҗеҜ№иұЎпјҲsourceпјүзҡ„жүҖжңүеҸҜжһҡдёҫеұһжҖ§пјҢеӨҚеҲ¶еҲ°зӣ®ж ҮеҜ№иұЎпјҲtargetпјүгҖӮиҜҘж–№жі•зҡ„з¬¬дёҖдёӘеҸӮж•°жҳҜзӣ®ж ҮеҜ№иұЎпјҢеҗҺйқўзҡ„еҸӮж•°йғҪжҳҜжәҗеҜ№иұЎгҖӮ

жіЁж„Ҹпјҡ

  • еҰӮжһңзӣ®ж ҮеҜ№иұЎдёҺжәҗеҜ№иұЎжңүеҗҢеҗҚеұһжҖ§пјҢжҲ–еӨҡдёӘжәҗеҜ№иұЎжңүеҗҢеҗҚеұһжҖ§пјҢеҲҷеҗҺйқўзҡ„еұһжҖ§дјҡиҰҶзӣ–еүҚйқўзҡ„еұһжҖ§гҖӮ
  • еҰӮжһңеҸӘжңүдёҖдёӘеҸӮж•°пјҢObject.assignдјҡзӣҙжҺҘиҝ”еӣһиҜҘеҸӮж•°гҖӮеҰӮжһңиҜҘеҸӮж•°дёҚжҳҜеҜ№иұЎпјҢеҲҷдјҡе…ҲиҪ¬жҲҗеҜ№иұЎпјҢз„¶еҗҺиҝ”еӣһгҖӮз”ұдәҺundefinedе’Ңnullж— жі•иҪ¬жҲҗеҜ№иұЎпјҢжүҖд»ҘеҰӮжһңе®ғд»¬дҪңдёәеҸӮж•°пјҢе°ұдјҡжҠҘй”ҷгҖӮ
  • еҰӮжһңйқһеҜ№иұЎеҸӮж•°еҮәзҺ°еңЁжәҗеҜ№иұЎзҡ„дҪҚзҪ®пјҲеҚійқһйҰ–еҸӮж•°пјүпјҢйӮЈд№ҲеӨ„зҗҶи§„еҲҷжңүжүҖдёҚеҗҢгҖӮйҰ–е…ҲпјҢиҝҷдәӣеҸӮж•°йғҪдјҡиҪ¬жҲҗеҜ№иұЎпјҢеҰӮжһңж— жі•иҪ¬жҲҗеҜ№иұЎпјҢе°ұдјҡи·іиҝҮгҖӮиҝҷж„Ҹе‘ізқҖпјҢеҰӮжһңundefinedе’ҢnullдёҚеңЁйҰ–еҸӮж•°пјҢе°ұдёҚдјҡжҠҘй”ҷгҖӮе…¶д»–зұ»еһӢзҡ„еҖјпјҲеҚіж•°еҖјгҖҒеӯ—з¬ҰдёІе’Ңеёғе°”еҖјпјүдёҚеңЁйҰ–еҸӮж•°пјҢд№ҹдёҚдјҡжҠҘй”ҷгҖӮдҪҶжҳҜпјҢйҷӨдәҶеӯ—з¬ҰдёІдјҡд»Ҙж•°з»„еҪўејҸпјҢжӢ·иҙқе…Ҙзӣ®ж ҮеҜ№иұЎпјҢе…¶д»–еҖјйғҪдёҚдјҡдә§з”ҹж•ҲжһңгҖӮ
  • Object.assignжӢ·иҙқзҡ„еұһжҖ§жҳҜжңүйҷҗеҲ¶зҡ„пјҢеҸӘжӢ·иҙқжәҗеҜ№иұЎзҡ„иҮӘиә«еұһжҖ§пјҲдёҚжӢ·иҙқз»§жүҝеұһжҖ§пјүпјҢд№ҹдёҚжӢ·иҙқдёҚеҸҜжһҡдёҫзҡ„еұһжҖ§пјҲenumerable: falseпјүгҖӮ
  • еұһжҖ§еҗҚдёә Symbol еҖјзҡ„еұһжҖ§пјҢд№ҹдјҡиў«Object.assignжӢ·иҙқгҖӮ
var target = { a: 1, b: 1 };
var source1 = { b: 2, c: 2 };
var source2 = { c: 3 };
Object.assign(target, source1, source2);
target // {a:1, b:2, c:3}
Object.assign(target) === target // true
typeof Object.assign(2) // "object"

Object.assign({b: 'c'},
  Object.defineProperty({}, 'invisible', {
    enumerable: false,
    value: 'hello'
  })
)
// { b: 'c' }
Object.assign({ a: 'b' }, { [Symbol('c')]: 'd' })
// { a: 'b', Symbol(c): 'd' }

# 2.1 жіЁж„ҸзӮ№

1гҖҒжө…жӢ·иҙқ

Object.assignж–№жі•е®һиЎҢзҡ„жҳҜжө…жӢ·иҙқпјҢиҖҢдёҚжҳҜж·ұжӢ·иҙқгҖӮд№ҹе°ұжҳҜиҜҙпјҢеҰӮжһңжәҗеҜ№иұЎжҹҗдёӘеұһжҖ§зҡ„еҖјжҳҜеҜ№иұЎпјҢйӮЈд№Ҳзӣ®ж ҮеҜ№иұЎжӢ·иҙқеҫ—еҲ°зҡ„жҳҜиҝҷдёӘеҜ№иұЎзҡ„еј•з”ЁгҖӮ

const obj1 = {a: {b: 1}};
const obj2 = Object.assign({}, obj1);
obj1.a.b = 2;
obj2.a.b // 2

2гҖҒеҗҢеҗҚеұһжҖ§зҡ„жӣҝжҚў

еҜ№дәҺиҝҷз§ҚеөҢеҘ—зҡ„еҜ№иұЎпјҢдёҖж—ҰйҒҮеҲ°еҗҢеҗҚеұһжҖ§пјҢObject.assignзҡ„еӨ„зҗҶж–№жі•жҳҜжӣҝжҚўпјҢиҖҢдёҚжҳҜж·»еҠ гҖӮ

const target = { a: { b: 'c', d: 'e' } }
const source = { a: { b: 'hello' } }
// targetдёӯaеұһжҖ§иў«ж•ҙдёӘжӣҝжҚўжҺүдәҶпјҢиҝҷйҖҡеёёдёҚжҳҜејҖеҸ‘иҖ…жғіиҰҒзҡ„пјҢйңҖиҰҒзү№еҲ«е°ҸеҝғгҖӮ
Object.assign(target, source)
// { a: { b: 'hello' } }

дёҖдәӣеҮҪж•°еә“жҸҗдҫӣObject.assignзҡ„е®ҡеҲ¶зүҲжң¬пјҲжҜ”еҰӮ Lodash зҡ„_.defaultsDeepж–№жі•пјүпјҢеҸҜд»Ҙеҫ—еҲ°ж·ұжӢ·иҙқзҡ„еҗҲе№¶гҖӮ

3гҖҒж•°з»„зҡ„еӨ„зҗҶ

Object.assignеҸҜд»Ҙз”ЁжқҘеӨ„зҗҶж•°з»„пјҢдҪҶжҳҜдјҡжҠҠж•°з»„и§ҶдёәеҜ№иұЎгҖӮ

Object.assign([1, 2, 3], [4, 5])
// [4, 5, 3]

дёҠйқўд»Јз ҒдёӯпјҢObject.assignжҠҠж•°з»„и§ҶдёәеұһжҖ§еҗҚдёә 0гҖҒ1гҖҒ2 зҡ„еҜ№иұЎпјҢеӣ жӯӨжәҗж•°з»„зҡ„ 0 еҸ·еұһжҖ§4иҰҶзӣ–дәҶзӣ®ж Үж•°з»„зҡ„ 0 еҸ·еұһжҖ§1гҖӮ

4гҖҒеҸ–еҖјеҮҪж•°зҡ„еӨ„зҗҶ

Object.assignеҸӘиғҪиҝӣиЎҢеҖјзҡ„еӨҚеҲ¶пјҢеҰӮжһңиҰҒеӨҚеҲ¶зҡ„еҖјжҳҜдёҖдёӘеҸ–еҖјеҮҪж•°пјҢйӮЈд№Ҳе°ҶжұӮеҖјеҗҺеҶҚеӨҚеҲ¶гҖӮ

const source = {
  get foo() { return 1 }
};
const target = {};
Object.assign(target, source)
// { foo: 1 }

дёҠйқўд»Јз ҒдёӯпјҢsourceеҜ№иұЎзҡ„fooеұһжҖ§жҳҜдёҖдёӘеҸ–еҖјеҮҪж•°пјҢObject.assignдёҚдјҡеӨҚеҲ¶иҝҷдёӘеҸ–еҖјеҮҪж•°пјҢеҸӘдјҡжӢҝеҲ°еҖјд»ҘеҗҺпјҢе°ҶиҝҷдёӘеҖјеӨҚеҲ¶иҝҮеҺ»гҖӮ

# 2.2 еёёи§Ғз”ЁйҖ”

1гҖҒдёәеҜ№иұЎж·»еҠ еұһжҖ§

class Point {
  constructor(x, y) {
    Object.assign(this, {x, y});
  }
}

дёҠйқўж–№жі•йҖҡиҝҮObject.assignж–№жі•пјҢе°ҶxеұһжҖ§е’ҢyеұһжҖ§ж·»еҠ еҲ°Pointзұ»зҡ„еҜ№иұЎе®һдҫӢгҖӮ

2гҖҒдёәеҜ№иұЎж·»еҠ ж–№жі•

Object.assign(SomeClass.prototype, {
  someMethod(arg1, arg2) { },
  anotherMethod() { }
});
// зӯүеҗҢдәҺдёӢйқўзҡ„еҶҷжі•
SomeClass.prototype.someMethod = function (arg1, arg2) { };
SomeClass.prototype.anotherMethod = function () { };

дёҠйқўд»Јз ҒдҪҝз”ЁдәҶеҜ№иұЎеұһжҖ§зҡ„з®ҖжҙҒиЎЁзӨәжі•пјҢзӣҙжҺҘе°ҶдёӨдёӘеҮҪж•°ж”ҫеңЁеӨ§жӢ¬еҸ·дёӯпјҢеҶҚдҪҝз”Ёassignж–№жі•ж·»еҠ еҲ°SomeClass.prototypeд№ӢдёӯгҖӮ

3гҖҒе…ӢйҡҶеҜ№иұЎ

function clone(origin) {
  return Object.assign({}, origin);
}

дёҠйқўд»Јз Ғе°ҶеҺҹе§ӢеҜ№иұЎжӢ·иҙқеҲ°дёҖдёӘз©әеҜ№иұЎпјҢе°ұеҫ—еҲ°дәҶеҺҹе§ӢеҜ№иұЎзҡ„е…ӢйҡҶгҖӮ

дёҚиҝҮпјҢйҮҮз”Ёиҝҷз§Қж–№жі•е…ӢйҡҶпјҢеҸӘиғҪе…ӢйҡҶеҺҹе§ӢеҜ№иұЎиҮӘиә«зҡ„еҖјпјҢдёҚиғҪе…ӢйҡҶе®ғз»§жүҝзҡ„еҖјгҖӮеҰӮжһңжғіиҰҒдҝқжҢҒз»§жүҝй“ҫпјҢеҸҜд»ҘйҮҮз”ЁдёӢйқўзҡ„д»Јз ҒгҖӮ

function clone(origin) {
  let originProto = Object.getPrototypeOf(origin);
  return Object.assign(Object.create(originProto), origin);
}

4гҖҒеҗҲе№¶еҜ№дёӘеҜ№иұЎ

е°ҶеӨҡдёӘеҜ№иұЎеҗҲе№¶еҲ°жҹҗдёӘеҜ№иұЎгҖӮ

const merge =
  (target, ...sources) => Object.assign(target, ...sources);

еҰӮжһңеёҢжңӣеҗҲе№¶еҗҺиҝ”еӣһдёҖдёӘж–°еҜ№иұЎпјҢеҸҜд»Ҙж”№еҶҷдёҠйқўеҮҪж•°пјҢеҜ№дёҖдёӘз©әеҜ№иұЎеҗҲе№¶гҖӮ

const merge =
  (...sources) => Object.assign({}, ...sources);

5гҖҒдёәеұһжҖ§жҢҮе®ҡй»ҳи®ӨеҖј

const DEFAULTS = {
  logLevel: 0,
  outputFormat: 'html'
};

function processContent(options) {
  options = Object.assign({}, DEFAULTS, options);
  console.log(options);
  // ...
}

дёҠйқўд»Јз ҒдёӯпјҢDEFAULTSеҜ№иұЎжҳҜй»ҳи®ӨеҖјпјҢoptionsеҜ№иұЎжҳҜз”ЁжҲ·жҸҗдҫӣзҡ„еҸӮж•°гҖӮObject.assignж–№жі•е°ҶDEFAULTSе’ҢoptionsеҗҲе№¶жҲҗдёҖдёӘж–°еҜ№иұЎпјҢеҰӮжһңдёӨиҖ…жңүеҗҢеҗҚеұһжҖ§пјҢеҲҷoptionsзҡ„еұһжҖ§еҖјдјҡиҰҶзӣ–DEFAULTSзҡ„еұһжҖ§еҖјгҖӮ

жіЁж„ҸпјҢз”ұдәҺеӯҳеңЁжө…жӢ·иҙқзҡ„й—®йўҳпјҢDEFAULTSеҜ№иұЎе’ҢoptionsеҜ№иұЎзҡ„жүҖжңүеұһжҖ§зҡ„еҖјпјҢжңҖеҘҪйғҪжҳҜз®ҖеҚ•зұ»еһӢпјҢдёҚиҰҒжҢҮеҗ‘еҸҰдёҖдёӘеҜ№иұЎгҖӮеҗҰеҲҷпјҢDEFAULTSеҜ№иұЎзҡ„иҜҘеұһжҖ§еҫҲеҸҜиғҪдёҚиө·дҪңз”ЁгҖӮ

const DEFAULTS = {
  url: {
    host: 'example.com',
    port: 7070
  },
};

processContent({ url: {port: 8000} })
// {
//   url: {port: 8000}
// }

дёҠйқўд»Јз Ғзҡ„еҺҹж„ҸжҳҜе°Ҷurl.portж”№жҲҗ 8000пјҢurl.hostдёҚеҸҳгҖӮе®һйҷ…з»“жһңеҚҙжҳҜoptions.urlиҰҶзӣ–жҺүDEFAULTS.urlпјҢжүҖд»Ҙurl.hostе°ұдёҚеӯҳеңЁдәҶгҖӮ

# 3. Object.keys()пјҢObject.values()пјҢObject.entries()

# 3.1 Object.keys()

ES5 еј•е…ҘдәҶObject.keysж–№жі•пјҢиҝ”еӣһдёҖдёӘж•°з»„пјҢжҲҗе‘ҳжҳҜеҸӮж•°еҜ№иұЎиҮӘиә«зҡ„пјҲдёҚеҗ«з»§жүҝзҡ„пјүжүҖжңүеҸҜйҒҚеҺҶпјҲenumerableпјүеұһжҖ§зҡ„й”®еҗҚгҖӮ

const obj = { foo: 'bar', baz: 42, age: [14, 16, 18, 20] };
Object.keys(obj)
// ['foo', 'baz', 'age']

ES2017 еј•е…ҘдәҶи·ҹObject.keysй…ҚеҘ—зҡ„Object.valuesе’ҢObject.entriesпјҢдҪңдёәйҒҚеҺҶдёҖдёӘеҜ№иұЎзҡ„иЎҘе……жүӢж®өпјҢдҫӣfor...ofеҫӘзҺҜдҪҝз”ЁгҖӮ

let {keys, values, entries} = Object;
let obj = { a: 1, b: 2, c: 3 };
for (let key of keys(obj)) {
  console.log(key); // 'a', 'b', 'c'
}
for (let value of values(obj)) {
  console.log(value); // 1, 2, 3
}
for (let [key, value] of entries(obj)) {
  console.log([key, value]); // ['a', 1], ['b', 2], ['c', 3]
}

# 3.2 Object.values()

Object.valuesж–№жі•иҝ”еӣһдёҖдёӘз»ҷе®ҡеҜ№иұЎиҮӘиә«(дёҚеҗ«з»§жүҝзҡ„)зҡ„жүҖжңүеҸҜжһҡдёҫеұһжҖ§еҖјзҡ„ж•°з»„гҖӮиҜҘж–№жі•жҸҗдҫӣдәҶдёҖз§Қз®ҖжҙҒзҡ„ж–№ејҸжқҘиҺ·еҸ–еҜ№иұЎзҡ„еҖјйӣҶеҗҲпјҢиҖҢдёҚйңҖиҰҒжүӢеҠЁйҒҚеҺҶеҜ№иұЎгҖӮ

Object.valuesдјҡиҝҮж»ӨеұһжҖ§еҗҚдёә Symbol еҖјзҡ„еұһжҖ§гҖӮ

еҰӮжһңObject.valuesж–№жі•зҡ„еҸӮж•°жҳҜдёҖдёӘеӯ—з¬ҰдёІпјҢдјҡиҝ”еӣһеҗ„дёӘеӯ—з¬Ұз»„жҲҗзҡ„дёҖдёӘж•°з»„гҖӮ

еҰӮжһңеҸӮж•°дёҚжҳҜеҜ№иұЎпјҢObject.valuesдјҡе…Ҳе°Ҷе…¶иҪ¬дёәеҜ№иұЎгҖӮз”ұдәҺж•°еҖје’Ңеёғе°”еҖјзҡ„еҢ…иЈ…еҜ№иұЎпјҢйғҪдёҚдјҡдёәе®һдҫӢж·»еҠ йқһз»§жүҝзҡ„еұһжҖ§гҖӮжүҖд»ҘпјҢObject.valuesдјҡиҝ”еӣһз©әж•°з»„гҖӮ

const obj = { foo: 'bar', baz: 42, age: [14, 16, 18, 20] };
Object.values(obj)
// ["bar", 42, [14, 16, 18, 20]]

// дёҚдјҡиҝ”еӣһйӣҶжҲҗзҡ„еұһжҖ§еҖј
const obj = Object.create({}, {p: {value: 42}});
Object.values(obj) // []
// еұһжҖ§еҗҚдёә Symbol еҖј дјҡеҝҪз•Ҙ
Object.values({ [Symbol()]: 123, foo: 'abc' }); // ['abc']
// еҸӮж•°жҳҜеӯ—з¬ҰдёІ
Object.values('foo') // ['f', 'o', 'o']

Object.values(42) // []

# 3.3 Object.entries()

Object.entries()ж–№жі•иҝ”еӣһдёҖдёӘж•°з»„пјҢжҲҗе‘ҳжҳҜеҸӮж•°еҜ№иұЎиҮӘиә«зҡ„пјҲдёҚеҗ«з»§жүҝзҡ„пјүжүҖжңүеҸҜйҒҚеҺҶпјҲenumerableпјүеұһжҖ§зҡ„й”®еҖјеҜ№ж•°з»„гҖӮйҷӨдәҶиҝ”еӣһеҖјдёҚдёҖж ·пјҢиҜҘж–№жі•зҡ„иЎҢдёәдёҺObject.valuesеҹәжң¬дёҖиҮҙгҖӮ

const obj = { foo: 'bar', baz: 42, age: [14, 16, 18, 20] };
Object.entries(obj) 
// [
//   ["foo", "bar"],
//   ["baz", 42],
//   ['age', [14, 16, 18, 20]],
// ]
// еҺҹеҜ№иұЎзҡ„еұһжҖ§еҗҚжҳҜдёҖдёӘ Symbol еҖјпјҢиҜҘеұһжҖ§дјҡиў«еҝҪз•Ҙ
Object.entries({ [Symbol()]: 123, foo: 'abc' });
// [ [ 'foo', 'abc' ] ]

Object.entriesзҡ„еҹәжң¬з”ЁйҖ”жҳҜйҒҚеҺҶеҜ№иұЎзҡ„еұһжҖ§гҖӮ

let obj = { one: 1, two: 2 };
for (let [k, v] of Object.entries(obj)) {
  console.log(`${JSON.stringify(k)}: ${JSON.stringify(v)}`);
}
// "one": 1
// "two": 2

Object.entriesж–№жі•зҡ„еҸҰдёҖдёӘз”ЁеӨ„жҳҜпјҢе°ҶеҜ№иұЎиҪ¬дёәзңҹжӯЈзҡ„Mapз»“жһ„гҖӮ

const obj = { foo: 'bar', baz: 42 };
const map = new Map(Object.entries(obj));
map // Map { foo: "bar", baz: 42 }

иҮӘе·ұе®һзҺ°Object.entriesж–№жі•пјҢйқһеёёз®ҖеҚ•гҖӮ

// GeneratorеҮҪж•°зҡ„зүҲжң¬
function* entries(obj) {
  for (let key of Object.keys(obj)) {
    yield [key, obj[key]];
  }
}
// йқһGeneratorеҮҪж•°зҡ„зүҲжң¬
function entries(obj) {
  let arr = [];
  for (let key of Object.keys(obj)) {
    arr.push([key, obj[key]]);
  }
  return arr;
}

# 4. Object.fromEntries()

Object.fromEntries()ж–№жі•жҳҜObject.entries()зҡ„йҖҶж“ҚдҪңпјҢз”ЁдәҺе°ҶдёҖдёӘй”®еҖјеҜ№ж•°з»„иҪ¬дёәеҜ№иұЎгҖӮ

Object.fromEntries([
  ['foo', 'bar'],
  ['baz', 42]
])
// { foo: "bar", baz: 42 }

иҜҘж–№жі•зҡ„дё»иҰҒзӣ®зҡ„пјҢжҳҜе°Ҷй”®еҖјеҜ№зҡ„ж•°жҚ®з»“жһ„иҝҳеҺҹдёәеҜ№иұЎпјҢеӣ жӯӨзү№еҲ«йҖӮеҗҲе°Ҷ Map з»“жһ„иҪ¬дёәеҜ№иұЎгҖӮ

// дҫӢдёҖ
const entries = new Map([
  ['foo', 'bar'],
  ['baz', 42]
]);
Object.fromEntries(entries)
// { foo: "bar", baz: 42 }

// дҫӢдәҢ
const map = new Map().set('foo', true).set('bar', false);
Object.fromEntries(map)
// { foo: true, bar: false }

иҜҘж–№жі•зҡ„дёҖдёӘз”ЁеӨ„жҳҜй…ҚеҗҲURLSearchParamsеҜ№иұЎпјҢе°ҶжҹҘиҜўеӯ—з¬ҰдёІиҪ¬дёәеҜ№иұЎгҖӮ

Object.fromEntries(new URLSearchParams('foo=bar&baz=qux'))
// { foo: "bar", baz: "qux" }

# 5. Object.getOwnPropertyDescriptors()

ES5 зҡ„Object.getOwnPropertyDescriptor()ж–№жі•дјҡиҝ”еӣһжҹҗдёӘеҜ№иұЎеұһжҖ§зҡ„жҸҸиҝ°еҜ№иұЎпјҲdescriptorпјүгҖӮ

ES2017 еј•е…ҘдәҶObject.getOwnPropertyDescriptors()ж–№жі•пјҢиҝ”еӣһжҢҮе®ҡеҜ№иұЎжүҖжңүиҮӘиә«еұһжҖ§пјҲйқһз»§жүҝеұһжҖ§пјүзҡ„жҸҸиҝ°еҜ№иұЎгҖӮ

const obj = {
  foo: 123,
  get bar() { return 'abc' }
};

Object.getOwnPropertyDescriptors(obj)
// { foo:
//    {value: 123, writable: true, enumerable: true, configurable: true},
//   bar:
//    {set: undefined, enumerable: true, configurable: true, get: Ж’}

иҜҘж–№жі•зҡ„е®һзҺ°йқһеёёе®№жҳ“гҖӮ

function getOwnPropertyDescriptors(obj) {
  const result = {};
  for (let key of Reflect.ownKeys(obj)) {
    result[key] = Object.getOwnPropertyDescriptor(obj, key);
  }
  return result;
}

иҜҘж–№жі•зҡ„еј•е…Ҙзӣ®зҡ„пјҢдё»иҰҒжҳҜдёәдәҶи§ЈеҶіObject.assign()ж— жі•жӯЈзЎ®жӢ·иҙқgetеұһжҖ§е’ҢsetеұһжҖ§зҡ„й—®йўҳгҖӮ

const source = {
  set foo(value) {
    console.log(value);
  }
};

const target1 = {};
Object.assign(target1, source);

Object.getOwnPropertyDescriptor(target1, 'foo')
// { value: undefined,
//   writable: true,
//   enumerable: true,
//   configurable: true }

дёҠйқўд»Јз ҒдёӯпјҢsourceеҜ№иұЎзҡ„fooеұһжҖ§зҡ„еҖјжҳҜдёҖдёӘиөӢеҖјеҮҪж•°пјҢObject.assignж–№жі•е°ҶиҝҷдёӘеұһжҖ§жӢ·иҙқз»ҷtarget1еҜ№иұЎпјҢз»“жһңиҜҘеұһжҖ§зҡ„еҖјеҸҳжҲҗдәҶundefinedгҖӮиҝҷжҳҜеӣ дёәObject.assignж–№жі•жҖ»жҳҜжӢ·иҙқдёҖдёӘеұһжҖ§зҡ„еҖјпјҢиҖҢдёҚдјҡжӢ·иҙқе®ғиғҢеҗҺзҡ„иөӢеҖјж–№жі•жҲ–еҸ–еҖјж–№жі•гҖӮ

иҝҷж—¶пјҢObject.getOwnPropertyDescriptors()ж–№жі•й…ҚеҗҲObject.defineProperties()ж–№жі•пјҢе°ұеҸҜд»Ҙе®һзҺ°жӯЈзЎ®жӢ·иҙқгҖӮ

const source = {
  set foo(value) {
    console.log(value);
  }
};

const target2 = {};
Object.defineProperties(target2, Object.getOwnPropertyDescriptors(source));
Object.getOwnPropertyDescriptor(target2, 'foo')
// { get: undefined,
//   set: [Function: set foo],
//   enumerable: true,
//   configurable: true }

дёҠйқўд»Јз ҒдёӯпјҢдёӨдёӘеҜ№иұЎеҗҲе№¶зҡ„йҖ»иҫ‘еҸҜд»ҘеҶҷжҲҗдёҖдёӘеҮҪж•°гҖӮ

const shallowMerge = (target, source) => Object.defineProperties(
  target,
  Object.getOwnPropertyDescriptors(source)
);

Object.getOwnPropertyDescriptors()ж–№жі•зҡ„еҸҰдёҖдёӘз”ЁеӨ„пјҢжҳҜй…ҚеҗҲObject.create()ж–№жі•пјҢе°ҶеҜ№иұЎеұһжҖ§е…ӢйҡҶеҲ°дёҖдёӘж–°еҜ№иұЎгҖӮиҝҷеұһдәҺжө…жӢ·иҙқгҖӮ

const clone = Object.create(Object.getPrototypeOf(obj),
  Object.getOwnPropertyDescriptors(obj));

// жҲ–иҖ…

const shallowClone = (obj) => Object.create(
  Object.getPrototypeOf(obj),
  Object.getOwnPropertyDescriptors(obj)
);

дёҠйқўд»Јз Ғдјҡе…ӢйҡҶеҜ№иұЎobjгҖӮ

еҸҰеӨ–пјҢObject.getOwnPropertyDescriptors()ж–№жі•еҸҜд»Ҙе®һзҺ°дёҖдёӘеҜ№иұЎз»§жүҝеҸҰдёҖдёӘеҜ№иұЎгҖӮд»ҘеүҚпјҢз»§жүҝеҸҰдёҖдёӘеҜ№иұЎпјҢеёёеёёеҶҷжҲҗдёӢйқўиҝҷж ·гҖӮ

const obj = {
  __proto__: prot,
  foo: 123,
};

ES6 и§„е®ҡ__proto__еҸӘжңүжөҸи§ҲеҷЁиҰҒйғЁзҪІпјҢе…¶д»–зҺҜеўғдёҚз”ЁйғЁзҪІгҖӮеҰӮжһңеҺ»йҷӨ__proto__пјҢдёҠйқўд»Јз Ғе°ұиҰҒж”№жҲҗдёӢйқўиҝҷж ·гҖӮ

const obj = Object.create(prot);
obj.foo = 123;

// жҲ–иҖ…

const obj = Object.assign(
  Object.create(prot),
  {
    foo: 123,
  }
);

жңүдәҶObject.getOwnPropertyDescriptors()пјҢжҲ‘д»¬е°ұжңүдәҶеҸҰдёҖз§ҚеҶҷжі•гҖӮ

const obj = Object.create(
  prot,
  Object.getOwnPropertyDescriptors({
    foo: 123,
  })
);

Object.getOwnPropertyDescriptors()д№ҹеҸҜд»Ҙз”ЁжқҘе®һзҺ° MixinпјҲж··е…ҘпјүжЁЎејҸгҖӮ

let mix = (object) => ({
  with: (...mixins) => mixins.reduce(
    (c, mixin) => Object.create(
      c, Object.getOwnPropertyDescriptors(mixin)
    ), object)
});

// multiple mixins example
let a = {a: 'a'};
let b = {b: 'b'};
let c = {c: 'c'};
let d = mix(c).with(a, b);

d.c // "c"
d.b // "b"
d.a // "a"

дёҠйқўд»Јз Ғиҝ”еӣһдёҖдёӘж–°зҡ„еҜ№иұЎdпјҢд»ЈиЎЁдәҶеҜ№иұЎaе’Ңbиў«ж··е…ҘдәҶеҜ№иұЎcзҡ„ж“ҚдҪңгҖӮ

еҮәдәҺе®Ңж•ҙжҖ§зҡ„иҖғиҷ‘пјҢObject.getOwnPropertyDescriptors()иҝӣе…Ҙж ҮеҮҶд»ҘеҗҺпјҢд»ҘеҗҺиҝҳдјҡж–°еўһReflect.getOwnPropertyDescriptors()ж–№жі•гҖӮ

# 6. __proto__еұһжҖ§пјҢObject.setPrototypeOf()пјҢObject.getPrototypeOf()

# 6.1 __proto__еұһжҖ§

__proto__еұһжҖ§пјҲеүҚеҗҺеҗ„дёӨдёӘдёӢеҲ’зәҝпјүпјҢз”ЁжқҘиҜ»еҸ–жҲ–и®ҫзҪ®еҪ“еүҚеҜ№иұЎзҡ„еҺҹеһӢеҜ№иұЎпјҲprototypeпјүгҖӮзӣ®еүҚпјҢжүҖжңүжөҸи§ҲеҷЁпјҲеҢ…жӢ¬ IE11пјүйғҪйғЁзҪІдәҶиҝҷдёӘеұһжҖ§гҖӮ

// es5 зҡ„еҶҷжі•
const obj = {
  method: function() { ... }
};
obj.__proto__ = someOtherObj;

// es6 зҡ„еҶҷжі•
var obj = Object.create(someOtherObj);
obj.method = function() { ... };

иҜҘеұһжҖ§жІЎжңүеҶҷе…Ҙ ES6 зҡ„жӯЈж–ҮпјҢиҖҢжҳҜеҶҷе…ҘдәҶйҷ„еҪ•пјҢеҺҹеӣ жҳҜ__proto__еүҚеҗҺзҡ„еҸҢдёӢеҲ’зәҝпјҢиҜҙжҳҺе®ғжң¬иҙЁдёҠжҳҜдёҖдёӘеҶ…йғЁеұһжҖ§пјҢиҖҢдёҚжҳҜдёҖдёӘжӯЈејҸзҡ„еҜ№еӨ–зҡ„ APIпјҢеҸӘжҳҜз”ұдәҺжөҸи§ҲеҷЁе№ҝжіӣж”ҜжҢҒпјҢжүҚиў«еҠ е…ҘдәҶ ES6гҖӮж ҮеҮҶжҳҺзЎ®и§„е®ҡпјҢеҸӘжңүжөҸи§ҲеҷЁеҝ…йЎ»йғЁзҪІиҝҷдёӘеұһжҖ§пјҢе…¶д»–иҝҗиЎҢзҺҜеўғдёҚдёҖе®ҡйңҖиҰҒйғЁзҪІпјҢиҖҢдё”ж–°зҡ„д»Јз ҒжңҖеҘҪи®ӨдёәиҝҷдёӘеұһжҖ§жҳҜдёҚеӯҳеңЁзҡ„гҖӮеӣ жӯӨпјҢж— и®әд»ҺиҜӯд№үзҡ„и§’еәҰпјҢиҝҳжҳҜд»Һе…је®№жҖ§зҡ„и§’еәҰпјҢйғҪдёҚиҰҒдҪҝз”ЁиҝҷдёӘеұһжҖ§пјҢиҖҢжҳҜдҪҝз”ЁдёӢйқўзҡ„Object.setPrototypeOf()пјҲеҶҷж“ҚдҪңпјүгҖҒObject.getPrototypeOf()пјҲиҜ»ж“ҚдҪңпјүгҖҒObject.create()пјҲз”ҹжҲҗж“ҚдҪңпјүд»ЈжӣҝгҖӮ

е®һзҺ°дёҠпјҢ__proto__и°ғз”Ёзҡ„жҳҜObject.prototype.__proto__пјҢе…·дҪ“е®һзҺ°еҰӮдёӢгҖӮ

Object.defineProperty(Object.prototype, '__proto__', {
  get() {
    let _thisObj = Object(this);
    return Object.getPrototypeOf(_thisObj);
  },
  set(proto) {
    if (this === undefined || this === null) {
      throw new TypeError();
    }
    if (!isObject(this)) {
      return undefined;
    }
    if (!isObject(proto)) {
      return undefined;
    }
    let status = Reflect.setPrototypeOf(this, proto);
    if (!status) {
      throw new TypeError();
    }
  },
});

function isObject(value) {
  return Object(value) === value;
}

еҰӮжһңдёҖдёӘеҜ№иұЎжң¬иә«йғЁзҪІдәҶ__proto__еұһжҖ§пјҢиҜҘеұһжҖ§зҡ„еҖје°ұжҳҜеҜ№иұЎзҡ„еҺҹеһӢгҖӮ

Object.getPrototypeOf({ __proto__: null }) // null

# 6.2 Object.setPrototypeOf()

Object.setPrototypeOfж–№жі•зҡ„дҪңз”ЁдёҺ__proto__зӣёеҗҢпјҢз”ЁжқҘи®ҫзҪ®дёҖдёӘеҜ№иұЎзҡ„еҺҹеһӢеҜ№иұЎпјҲprototypeпјүпјҢиҝ”еӣһеҸӮж•°еҜ№иұЎжң¬иә«гҖӮе®ғжҳҜ ES6 жӯЈејҸжҺЁиҚҗзҡ„и®ҫзҪ®еҺҹеһӢеҜ№иұЎзҡ„ж–№жі•гҖӮ

// ж јејҸ
Object.setPrototypeOf(object, prototype)

// з”Ёжі•
const o = Object.setPrototypeOf({}, null);

иҜҘж–№жі•зӯүеҗҢдәҺдёӢйқўзҡ„еҮҪж•°гҖӮ

function setPrototypeOf(obj, proto) {
  obj.__proto__ = proto;
  return obj;
}

еҰӮжһңз¬¬дёҖдёӘеҸӮж•°дёҚжҳҜеҜ№иұЎпјҢдјҡиҮӘеҠЁиҪ¬дёәеҜ№иұЎгҖӮдҪҶжҳҜз”ұдәҺиҝ”еӣһзҡ„иҝҳжҳҜз¬¬дёҖдёӘеҸӮж•°пјҢжүҖд»ҘиҝҷдёӘж“ҚдҪңдёҚдјҡдә§з”ҹд»»дҪ•ж•ҲжһңгҖӮ

Object.setPrototypeOf(1, {}) === 1 // true
Object.setPrototypeOf('foo', {}) === 'foo' // true
Object.setPrototypeOf(true, {}) === true // true

з”ұдәҺundefinedе’Ңnullж— жі•иҪ¬дёәеҜ№иұЎпјҢжүҖд»ҘеҰӮжһңз¬¬дёҖдёӘеҸӮж•°жҳҜundefinedжҲ–nullпјҢе°ұдјҡжҠҘй”ҷгҖӮ

Object.setPrototypeOf(undefined, {})
// TypeError: Object.setPrototypeOf called on null or undefined

Object.setPrototypeOf(null, {})
// TypeError: Object.setPrototypeOf called on null or undefined

# 6.3 Object.getPrototypeOf()

иҜҘж–№жі•дёҺObject.setPrototypeOfж–№жі•й…ҚеҘ—пјҢз”ЁдәҺиҜ»еҸ–дёҖдёӘеҜ№иұЎзҡ„еҺҹеһӢеҜ№иұЎгҖӮ

Object.getPrototypeOf(obj);

еҰӮжһңеҸӮж•°дёҚжҳҜеҜ№иұЎпјҢдјҡиў«иҮӘеҠЁиҪ¬дёәеҜ№иұЎгҖӮ

// зӯүеҗҢдәҺ Object.getPrototypeOf(Number(1))
Object.getPrototypeOf(1)
// Number {[[PrimitiveValue]]: 0}

// зӯүеҗҢдәҺ Object.getPrototypeOf(String('foo'))
Object.getPrototypeOf('foo')
// String {length: 0, [[PrimitiveValue]]: ""}

// зӯүеҗҢдәҺ Object.getPrototypeOf(Boolean(true))
Object.getPrototypeOf(true)
// Boolean {[[PrimitiveValue]]: false}

Object.getPrototypeOf(1) === Number.prototype // true
Object.getPrototypeOf('foo') === String.prototype // true
Object.getPrototypeOf(true) === Boolean.prototype // true

еҰӮжһңеҸӮж•°жҳҜundefinedжҲ–nullпјҢе®ғд»¬ж— жі•иҪ¬дёәеҜ№иұЎпјҢжүҖд»ҘдјҡжҠҘй”ҷгҖӮ

Object.getPrototypeOf(null)
// TypeError: Cannot convert undefined or null to object

Object.getPrototypeOf(undefined)
// TypeError: Cannot convert undefined or null to object
